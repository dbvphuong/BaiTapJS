// Dùng vòng lặp while, so sánh trung bình cộng của min và max với số cần tìm
function binarySearch(array, value) {
    var mid,
        min = 0,
        max = array.length - 1;
        count = 0;
    while (min <= max){
        mid = Math.floor((min + max) /2);
        if (array[mid] === value){
            count++ ;
            array.splice(mid,1);
        } // nếu mid = value thì ta đếm +1 và xóa số đó đi 
        else if (array[mid] < value){
            min = mid + 1;
        }
        else
            max = mid - 1;
    }
    return count;

}
